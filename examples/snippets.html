3. Conditionals

Scanner scan = new Scanner(System.in); System.out.print("Enter a test score (0-100): "); int score = scan.nextInt(); String grade; if (score >= 90) { grade = "A"; } else if (score >= 80) { grade = "B"; } else if (score >= 70) { grade = "C"; } else if (score >= 60) { grade = "D"; } else { grade = "F"; } System.out.println("Score " + score + " is a " + grade); scan.close();

4. Loops

// Print a multiplication table int size = 5; for (int i = 1; i <= size; i++) { for (int j = 1; j <= size; j++) { System.out.printf("%4d", i * j); } System.out.println(); }

5. Arrays

int[] numbers = {64, 34, 25, 12, 22, 11, 90}; System.out.println("Before: " + Arrays.toString(numbers)); // Selection sort for (int i = 0; i < numbers.length - 1; i++) { int minIdx = i; for (int j = i + 1; j < numbers.length; j++) { if (numbers[j] < numbers[minIdx]) { minIdx = j; } } int temp = numbers[minIdx]; numbers[minIdx] = numbers[i]; numbers[i] = temp; } System.out.println("After: " + Arrays.toString(numbers));

6. 2D Arrays

// Create and display a 2D array (matrix) int rows = 3; int cols = 4; int[][] matrix = new int[rows][cols]; // Fill with values int value = 1; for (int i = 0; i < rows; i++) { for (int j = 0; j < cols; j++) { matrix[i][j] = value++; } } // Display the matrix System.out.println("Matrix:"); for (int i = 0; i < rows; i++) { for (int j = 0; j < cols; j++) { System.out.printf("%4d", matrix[i][j]); } System.out.println(); } // Compute row sums System.out.println("\nRow sums:"); for (int i = 0; i < rows; i++) { int sum = 0; for (int j = 0; j < cols; j++) { sum += matrix[i][j]; } System.out.println(" Row " + i + ": " + sum); }

7. Methods & Recursion

public static boolean isPrime(int n) { if (n < 2) return false; for (int i = 2; i <= Math.sqrt(n); i++) { if (n % i == 0) return false; } return true; } public static int factorial(int n) { if (n <= 1) return 1; return n * factorial(n - 1); } public static void main(String[] args) { // Find primes up to 50 System.out.print("Primes: "); for (int i = 2; i <= 50; i++) { if (isPrime(i)) { System.out.print(i + " "); } } System.out.println(); // Print factorials for (int i = 1; i <= 10; i++) { System.out.println(i + "! = " + factorial(i)); } }

// Count word frequencies String text = "the cat sat on the mat the cat"; String[] words = text.split(" "); HashMap<String, Integer> freq = new HashMap<>(); for (String word : words) { if (freq.containsKey(word)) { freq.put(word, freq.get(word) + 1); } else { freq.put(word, 1); } } // Sort keys and display ArrayList<String> keys = new ArrayList<>(freq.keySet()); Collections.sort(keys); for (String key : keys) { System.out.println(key + ": " + freq.get(key)); }
